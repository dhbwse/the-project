\newenvironment{tabl}[2]{%
	\restorelines%
	\penalty-200%  % breakings!
	\vspace{1.5ex plus 3ex minus 0.5ex}%
	\hfil%
	\xdef\tablcaption{#2}%
	\tabular{#1}%
	%\rowstyle{\bfseries}%  % make first table line bold
	}{%
	\endtabular{}%
	\captionof{table}{\tablcaption}%
	\vspace{1ex plus 2ex minus 0.3ex}%
}

\newcommand{\figur}[3][0.5\textwidth]{%
	\penalty-200%  % breakings!
	\vspace{1.5ex plus 3ex minus 0.5ex}%
	\begin{minipage}{\textwidth}%
		\centering%
		\includegraphics[width=#1]{img/#2}%
		\captionof{figure}{#3}%
	\end{minipage}%
	\vspace{1ex plus 2ex minus 0.3ex}%
}
